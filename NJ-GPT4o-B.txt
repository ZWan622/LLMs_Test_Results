from Bio import Phylo
from Bio.Phylo.TreeConstruction import DistanceCalculator, DistanceTreeConstructor
from Bio.Seq import Seq
from Bio.SeqRecord import SeqRecord
from Bio.Align import MultipleSeqAlignment
import pandas as pd

# Define the sequences
sequences = [
    SeqRecord(Seq("CGGTTAGCCTATTCTTTCTCACGGGACTTCCCAAAGGTAAGCTTTTCAGG"), id="seq1"),
    SeqRecord(Seq("CGGTTAGCGTGTTCTTGCTCACGGGACCTGCGAAAGGTAAGCCTGTCACG"), id="seq2"),
    SeqRecord(Seq("AACCGTGAAACGACCCCCTCCTCATGCCTAACCAGCGCGAGCGGCCCGCG"), id="seq3"),
    SeqRecord(Seq("AAACATGCATCGTTCCTCCACTCATGGCCAACCAACCAGGGCGTCCCGGA"), id="seq4"),
    SeqRecord(Seq("ATGCGTGCCGAGTCCGACTCCGGGTCACCCGTTCACGTTAACGTCTCCGG"), id="seq5")
]

# Create a multiple sequence alignment
alignment = MultipleSeqAlignment(sequences)

# Calculate the pairwise distance matrix using "identity" metric
calculator = DistanceCalculator('identity')
distance_matrix = calculator.get_distance(alignment)

# Display the initial distance matrix
distance_df = pd.DataFrame(data=distance_matrix.matrix, 
                           index=distance_matrix.names, 
                           columns=distance_matrix.names)
print("Initial Distance Matrix:")
print(distance_df)

# Use the DistanceTreeConstructor to construct the tree using NJ
constructor = DistanceTreeConstructor(calculator, 'nj')
tree = constructor.build_tree(alignment)

# Export the tree in Newick format
newick_tree = tree.format("newick")
print("\nPhylogenetic Tree in Newick Format:")
print(newick_tree)
