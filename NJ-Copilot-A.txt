from Bio import AlignIO
from Bio import Phylo
from Bio.Align.Applications import ClustalwCommandline

# Save your sequences in a FASTA file
with open("sequences.fasta", "w") as file:
    file.write(">seq1\nCGGTTAGCCTATTCTTTCTCACGGGACTTCCCAAAGGTAAGCTTTTCAGG\n")
    file.write(">seq2\nCGGTTAGCGTGTTCTTGCTCACGGGACCTGCGAAAGGTAAGCCTGTCACG\n")
    file.write(">seq3\nAACCGTGAAACGACCCCCTCCTCATGCCTAACCAGCGCGAGCGGCCCGCG\n")
    file.write(">seq4\nAAACATGCATCGTTCCTCCACTCATGGCCAACCAACCAGGGCGTCCCGGA\n")
    file.write(">seq5\nATGCGTGCCGAGTCCGACTCCGGGTCACCCGTTCACGTTAACGTCTCCGG\n")

# Align the sequences using ClustalW
clustalw_cline = ClustalwCommandline("clustalw2", infile="sequences.fasta")
stdout, stderr = clustalw_cline()

# Read the alignment
align = AlignIO.read("sequences.aln", "clustal")

# Build the phylogenetic tree
tree = Phylo.read("sequences.dnd", "newick")

# Display the tree
Phylo.draw(tree)
You will need to have ClustalW installed and available in your PATH. This script will align the sequences and generate a phylogenetic tree that you can view. The Phylo.draw(tree) command will display the tree.
